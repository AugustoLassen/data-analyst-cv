\sectionTitle{Selected Projects}{\faFlask}

\begin{projects}

  \project
    {Size-Shape Difference Method}
    {Python, Statistical Morphology, Large Survey Data}
    {\website{https://doi.org/10.1051/0004-6361/202556010}{A\&A 706, A85 (2025)}}
    {Developed a quantitative morphological diagnostic to distinguish ram-pressure
     stripping from tidal interactions in galaxy images.
     Applied the method to real survey data (GASP) for the first time,
     validating it against mock galaxy simulations.}

  \project
    {Star Formation in Dwarf Irregular Galaxies}
    {Python, HST Imaging, Spectral Fitting, Multi-wavelength Analysis}
    {\website{https://doi.org/10.1093/mnras/staf1049}{MNRAS 541, 1028 (2025)}}
    {End-to-end analysis pipeline combining HST broadband photometry and MUSE
     integral-field spectroscopy to characterise $\sim$100 star-forming clumps.
     Derived physical properties (ages, masses, star-formation rates) via SED fitting
     and emission-line diagnostics.}

  \project
    {Bound Star Clusters at Cosmic Dawn}
    {Photometric Catalogues, Gravitational Lensing, Statistical Inference}
    {\website{https://doi.org/10.1038/s41586-024-07703-7}{Nature 632, 513 (2024)}}
    {Contributed to the detection and characterisation of gravitationally bound
     star clusters in a lensed galaxy at $z \approx 6$, published in \textit{Nature}.
     Role: photometric analysis and quality control of HST cluster catalogue.}

  \project
    {UNWind HPC Project}
    {Python, SLURM, Numerical Simulations, HPC}
    {\website{https://www.cineca.it}{CINECA, Italy --- 2025}}
    {Awarded HPC allocation at CINECA for the project
     \textit{"Identifying ram-pressure induced unwinding arms in cluster spirals"}.
     Responsible for pipeline design, data management, and post-processing scripts.}

\end{projects}
